package nc.unc.kevintrochon;

import java.io.IOException;
import java.util.ArrayList;
import java.util.List;

public class Resolveur implements ResolveurInterface {
  GrilleImpl grille ;
  List<Character> charsPossible = new ArrayList<>();

  public Resolveur(String file) {
    grille = new GrilleImpl(file);
    for (char c:grille.getTableauPossible()
         ) {
      charsPossible.add(c);
    }
    try {
      GrilleParser.parseur(file, this.grille);
    }catch (IOException exception){
      exception.getMessage();
    }
  }

  public boolean solveur(int position){
    try {
      if (position == (grille.getDimension() * grille.getDimension())) {
        return true;
      }
      int i = position / grille.getDimension();
      int j = position % grille.getDimension();
      if (grille.getValue(i, j) != GrilleImpl.EMPTY) {
        return solveur(position + 1);
      }
      for (char c : charsPossible) {
        if (grille.verifLigne(i,c) && grille.verifColonne(j,c) && grille.verifRegion(i,j,c)){
          grille.initialisation(i, j, c);
          if (solveur(position + 1)) {
            return true;
          }
        }
      }
      grille.initialisation(i, j, GrilleImpl.EMPTY);
      return false;
    }catch (HorsBornesException | CaractereInterditException exception){
      return false;
    }
  }

}
